The data can be found in the file clean_data.csv (n=225):

file<-read.csv("clean_data.csv", header = TRUE,stringsAsFactors = FALSE)

The following code assesses the accuracy of participant’s responses to the control trials: 2 participants were excluded prior to analyses because their performance to these trials did not reach the pre-established threshold of 80% accuracy (same exclusion criterion as in our previous studies).

# selection of control trials
control<-subset(file, condition %in% c("cE2", "cE3", "cP"))

# check of performance
check.control<-data_summary1(control, varname="Accuracy", groupnames=c("Subject"))
EXCLUSION<-c()
PERFORMANCE<-c()
for (i in check.control$Subject){
  if (check.control[check.control$Subject==i,]$Accuracy<80){
      EXCLUSION <- append(EXCLUSION,paste(check.control[check.control$Subject==i,]$Subject))
      PERFORMANCE <- append(PERFORMANCE,paste(check.control[check.control$Subject==i,]$Accuracy))}
  } 

# exclusion of subjects
control<-subset(control, Subject %notin% EXCLUSION)
file<-subset(file,Subject %notin% EXCLUSION)

# number of subjects included in the analyses

Responses to the target conditions, IS-only and SS-only, were analysed using the data analysis pipelines from Guha, Moitra, and Marty (2021) and Guha, Moitra, and Marty (2022) (see the registration detail for more information).

# Selection of the target conditions
file<-subset(file, Condition %in% c("SS-only", "IS-only"))

# Setting up the Subject and Item factors
file$Subject<-as.factor(file$Subject)
file$Item<-as.factor(file$Item)

# Model 1: All vs. Every vs. Each
## Coding
file$Condition<-as.factor(file$Condition)
contrasts(file$Condition) <- rbind(-.5, .5)
#colnames(contrasts(file$Condition)) <- levels(file$Condition)[2]
file$Quantifier<-factor(file$Quantifier, levels=c("Every","All", "Each"))
#contrasts(file$Quantifier)

## Model
model1<- glmer(Response~Quantifier*Condition+(1|Subject)+(1|Item), family = binomial(link=logit), data = file, control=glmerControl(optimizer="bobyqa",optCtrl=list(maxfun=100000),check.conv.singular = .makeCC(action = "ignore",  tol = 1e-4)))
